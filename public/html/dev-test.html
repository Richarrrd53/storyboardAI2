JSON 格式說明 / Schema Reference
支援兩種格式：扁平結構 (title/scenes) 或巢狀結構 (metadata/frames)。Camera 欄位可拆成 cameraType + cameraDetail，也可只給單一 camera 字串 — 系統會自動以第一個逗號切分為徽章與細節。圖片支援 URL 或 data:image/...;base64,...。 
// Format A — 扁平結構
{
  "title": "一隻狗狗在草地上奔跑",
  "style": "電影風格",
  "ratio": "橫向16:9",
  "exportDate": "2026-04-20",
  "scenes": [
    {
      "id": 1,
      "image": "https://example.com/shot1.jpg",
      "title": "場景描述...",
      "cameraType": "特寫鏡頭",
      "cameraDetail": "大光圈淺景深，低角度。",
      "prompt": "close-up of a golden retriever..."
    }
  ]
}

// Format B — 巢狀結構（與 Storyboard 工具匯出格式相容）
{
  "metadata": {
    "originalStory": "一隻狗狗在草地上奔跑",
    "style": "電影風格",
    "ratio": "橫向16:9",
    "exportTime": "2026-04-20T13:53:38.778Z"
  },
  "frames": [
    {
      "id": 1,
      "title": "場景描述...",
      "camera": "特寫鏡頭，大光圈淺景深，低角度。",
      "prompt": "close-up of a golden retriever...",
      "image": "data:image/jpeg;base64,..."
    }
  ]
}
